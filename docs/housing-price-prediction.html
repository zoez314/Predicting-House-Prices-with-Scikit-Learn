# ===============================
# Step 4: Train/Test Split
# ===============================
# Split dataset into training (80%) and test (20%)
train_set, test_set = train_test_split(df, test_size=0.2, random_state=42)

# Separate features (X) and target (y)
X_train = train_set.drop("MedHouseVal", axis=1)
y_train = train_set["MedHouseVal"]
X_test  = test_set.drop("MedHouseVal", axis=1)
y_test  = test_set["MedHouseVal"]

# ===============================
# Step 5: Baseline Model - Linear Regression
# ===============================
# Pipeline ensures consistent preprocessing (scaling) + modeling
lin_pipe = Pipeline([
    ("scaler", StandardScaler()),     # Standardize features
    ("linreg", LinearRegression())    # Linear Regression model
])

# Fit the model on training data
lin_pipe.fit(X_train, y_train)

# Predict on test data
y_pred_lr = lin_pipe.predict(X_test)

# Calculate performance metrics
rmse_lr = mean_squared_error(y_test, y_pred_lr, squared=False)
r2_lr = r2_score(y_test, y_pred_lr)

print(f"Linear Regression → RMSE: {rmse_lr:.3f}, R²: {r2_lr:.3f}")

# ===============================
# Step 6: Stronger Model - Random Forest
# ===============================
# Random Forest can capture nonlinear relationships better than Linear Regression
rf_pipe = Pipeline([
    ("scaler", StandardScaler()),  # Scaling (not strictly needed for trees, but safe)
    ("rf", RandomForestRegressor(n_estimators=200, random_state=42, n_jobs=-1))
])

# Fit the Random Forest model
rf_pipe.fit(X_train, y_train)

# Predict on test data
y_pred_rf = rf_pipe.predict(X_test)

# Calculate performance metrics
rmse_rf = mean_squared_error(y_test, y_pred_rf, squared=False)
r2_rf = r2_score(y_test, y_pred_rf)

print(f"Random Forest → RMSE: {rmse_rf:.3f}, R²: {r2_rf:.3f}")
